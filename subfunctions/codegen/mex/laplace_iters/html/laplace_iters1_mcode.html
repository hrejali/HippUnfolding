1function [LP, iter_change] = laplace_iters(source,sink,init,maxiters)
 2
 3change_threshold = 10^(-3);
 4
 5% iterative averaging filter to solve Laplace equation with greater
 6% accuracy than initial fast-marching
 7fg = (init(:)>=0);
 8
 9% %filter set-up (26 nearest neighbours)
10% hl=ones(3,3,3);
11% nelem=numel(hl);
12% hl=hl./nelem;
13% hl=hl+hl(2,2,2)./(nelem-1); hl(2,2,2)=0;
14% filter set-up (6 NN)
15hl = strel('sphere',1);
16hl = double(hl.Neighborhood);
17
18
19bg=(~source & ~sink & isnan(init));
20
21%set up vel
22vel=init; %geodist to initialize greymatter
23vel(source)=0;
24vel(sink)=1;
25vel(bg)=0; %must be insulated after filtering
26iter_change = zeros(1,maxiters);
27
28% apply filter
29iters = 0;
30while iters < maxiters %max iterations
31    iters = iters+1;
32    
33    velup = imfilter(vel,hl,'replicate','conv'); %apply averaging filter
34    
35    %insulate the grey matter so gradient doesn't pass between folds -
36    %inspired by ndnanfilter
37    insulate_correction = imfilter(double(~bg),hl,'replicate','conv');
38    velup = velup./insulate_correction;
39    velup(bg|source)=0;
40    velup(sink)=1;
41    
42    %stopping condition
43    iter_change(iters) = nansum(abs(vel(fg)-velup(fg))); %compute change from last iteration
44    vel = velup;
45    if iter_change(iters)<change_threshold
46        break
47    end
48end
49vel = vel./max(vel(:));
50vel(~fg) = nan;
51LP = vel;
52end
